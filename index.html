let recognition; let isRecording = false; // 現在、ユーザーの意思で録音中かどうかを管理する最優先フラグ function initSpeechRecognition() { window.SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!window.SpeechRecognition) { alert("お使いのブラウザは音声認識に対応していません。"); return; } // すでにインスタンスがある場合は使い回すか、新しく作り直す if (!recognition) { recognition = new SpeechRecognition(); } recognition.lang = 'ja-JP'; recognition.interimResults = true; recognition.continuous = true; // 文字起こしデータを受信したとき recognition.onresult = (event) => { let finalTranscript = ''; for (let i = event.resultIndex; i < event.results.length; ++i) { if (event.results[i].isFinal) { finalTranscript += event.results[i][0].transcript + '\n'; } } if (finalTranscript) { const textarea = document.getElementById('textarea'); textarea.value += finalTranscript; // 常に最新のテキストが見えるように最下部へスクロール textarea.scrollTop = textarea.scrollHeight; } }; // ★最重要：ブラウザ側が勝手に終了したときの処理 recognition.onend = () => { // ユーザーがストップボタンを押していないのに止まった場合、強制再起動 if (isRecording) { console.log("ブラウザによる自動停止を検知。ストップボタンが押されていないため、即座に再起動します。"); safeStart(); } else { console.log("ユーザーによって正常に停止されました。"); } }; // エラーが発生したときの処理 recognition.onerror = (event) => { console.error("音声認識エラーが発生しました:", event.error); // ネットワーク瞬断や無音状態によるエラー(no-speech)でも、録音中なら再起動を試みる if (isRecording) { // エラー直後は少し時間を空けてから再起動すると安定します（0.5秒後） setTimeout(() => { safeStart(); }, 500); } }; } // 安全に音声認識をスタートさせる関数（重複起動エラーを防ぐ） function safeStart() { try { recognition.start(); } catch (e) { // すでに動いているというエラー（VMまたはInvisibleな状態）などの対策 console.log("再起動を試みましたが、すでに起動中か準備中でした:", e.message); } } // ユーザーが「スタートボタン」を押したとき function startListening() { if (isRecording) return; // すでに起動中なら何もしない isRecording = true; initSpeechRecognition(); safeStart(); console.log("文字起こしを開始しました（ストップボタンを押すまで継続します）"); } // ユーザーが「ストップボタン」を押したとき function stopListening() { isRecording = false; // ここでフラグを折ることで、onendでの自動再起動を防ぐ if (recognition) { recognition.stop(); console.log("ユーザーの指示により文字起こしを停止しました。"); } }
